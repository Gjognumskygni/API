function appendOptionLast(id)
{
  var elOptNew = document.createElement('option');

  }
  catch(ex) {
    elSel.add(elOptNew); // IE only
  }
}

	function expandSubmenus(menuId)
    {
        var el = document.getElementById('submenu_' + menuId);
        if(el == null)
            return ;

        if(el.style.display == 'none')
        {
            el.style.display= 'block';

        else
        {
            el.style.display= 'none';

			var lh = document.getElementById('left-menu').offsetHeight + document.getElementById('search_box').offsetHeight;

			{

				rh = document.getElementById('main-h').offsetHeight;
				lh = lh + 135;
			}

			}

			{
